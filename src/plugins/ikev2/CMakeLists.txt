if(NOT OPENSSL_FOUND)
  message(WARNING "openssl headers not found - ikev2 plugin disabled")
  return()
endif()

include_directories(${OPENSSL_INCLUDE_DIR})
add_compile_definitions(OPENSSL_SUPPRESS_DEPRECATED)

add_vpp_plugin(ikev2
  SOURCES
  ikev2.c
  ikev2_api.c
  ikev2_cli.c
  ikev2_crypto.c
  ikev2_format.c
  ikev2_payload.c

  API_FILES
  ikev2_types.api
  ikev2.api

  API_TEST_SOURCES
  ikev2_test.c

  INSTALL_HEADERS
  ikev2.h
  ikev2_priv.h

  LINK_LIBRARIES
  ${OPENSSL_LIBRARIES}
)
